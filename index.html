  <!-- Custom Script -->

  <script>

    // Global variables to store data

    let budgetData = [];

    let chartOfAccounts = {};


    // Load budget data from CSV

    function loadBudgetData() {

      Papa.parse('budget.csv', {

        download: true,

        header: true,

        complete: function(results) {

          budgetData = results.data;

          populateBudgetTable();

          createBudgetChart();

        },

        error: function(err) {

          console.error('Error loading CSV:', err);

        }

      });

    }


    // Populate the budget table with CSV data

    function populateBudgetTable() {

      const tbody = document.getElementById('budgetTable').getElementsByTagName('tbody')[0];

      tbody.innerHTML = ''; // Clear any existing rows


      budgetData.forEach(item => {

        let row = document.createElement('tr');

        row.innerHTML = `

          <td>${item["Account Number"] || item["AccountNumber"]}</td>

          <td>${item["Description"]}</td>

          <td>${item["Prior Year Spending"]}</td>

          <td>${item["Current Year Budget"]}</td>

        `;

        tbody.appendChild(row);

      });

    }


    // Create a bar chart comparing prior year spending vs. current year budget

    function createBudgetChart() {

      // Prepare labels and data from the CSV

      const labels = budgetData.map(item => item["Account Number"] || item["AccountNumber"]);

      const priorData = budgetData.map(item => parseFloat(item["Prior Year Spending"]) || 0);

      const currentData = budgetData.map(item => parseFloat(item["Current Year Budget"]) || 0);


      const ctx = document.getElementById('budgetChart').getContext('2d');

      new Chart(ctx, {

        type: 'bar',

        data: {

          labels: labels,

          datasets: [

            {

              label: 'Prior Year Spending',

              backgroundColor: 'rgba(54, 162, 235, 0.5)',

              borderColor: 'rgba(54, 162, 235, 1)',

              data: priorData

            },

            {

              label: 'Current Year Budget',

              backgroundColor: 'rgba(255, 99, 132, 0.5)',

              borderColor: 'rgba(255, 99, 132, 1)',

              data: currentData

            }

          ]

        },

        options: {

          responsive: true,

          scales: {

            x: {

              stacked: false

            },

            y: {

              stacked: false,

              beginAtZero: true

            }

          }

        }

      });

    }


    // Load chart of accounts from JSON

    function loadChartOfAccounts() {

      fetch('chartOfAccounts.json')

        .then(response => response.json())

        .then(data => {

          chartOfAccounts = data;

          populateChartOfAccounts();

        })

        .catch(error => console.error('Error loading chart of accounts:', error));

    }


    // Populate the sidebar with chart of accounts data

    function populateChartOfAccounts() {

      const list = document.getElementById('chartOfAccountsList');

      // Assume chartOfAccounts is a key-value mapping or hierarchical object

      for (const key in chartOfAccounts) {

        if (chartOfAccounts.hasOwnProperty(key)) {

          const li = document.createElement('li');

          li.className = 'list-group-item';

          // For example, display the account number and its name

          li.innerText = key + ': ' + chartOfAccounts[key];

          list.appendChild(li);

        }

      }

    }


    // Initialize the portal once the page is loaded

    window.onload = function() {

      loadBudgetData();

      loadChartOfAccounts();

    };

  </script>
